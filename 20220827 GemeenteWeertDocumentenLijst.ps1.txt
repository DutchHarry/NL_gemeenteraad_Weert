<#
Genereert een lijst van de documenten van voorgaande vergaderingen van gemeente(raad) Weert van 2015 t/m 2022

Verander $outputDirectory en eventueel $outputFile

documenten voor komende vergaderingen onderaan

Alleen documenten, geen video en audio, maar daarvoor zou je script aan kunnen passen

#>

$outputDirectory = "S:\_ue\"
$outputFile = "20220827_GemeenteWeertDocumentenLijst.txt"
$baseurl = "https://gemeenteraad.weert.nl"
$output = -join("$outputDirectory","$outputFile")
$listCies = "Gemeenteraad","Raadscommissie-Samenleving-en-Inwoners","Raadscommissie-Ruimte-Economie","Raadscommissie-Middelen-en-Bestuur","Sprekersplein","Informatiebijeenkomst","College-van-B-W","Collegeadviescommissies","Agendacommissie","Fractievoorzittersoverleg","Auditcommissie","Werkgeverscommissie","Informatiebijeenkomst-Bedrijfsvoering-Inwoners","Informatiebijeenkomst-Ruimte","Raadscommissie-Bedrijfsvoering-Inwoners","Raadscommissie-Ruimte","Openbare-bijeenkomst-georganiseerd-door-derden","Digitale-talkshow-Omgevingsvisie-Weert"
$listCies |%{
  $Cie = $_
	$listYears = 2015,2016,2017,2018,2019,2020,2021,2022
	$listYears |%{
	  $Jaar = $_
	  $url = -join("$baseurl","/Vergaderingen/","$Cie","/","$Jaar")
    $ProgressPreference = 'SilentlyContinue'    # Subsequent calls do not display UI.
      $page = Invoke-Webrequest $url
    $ProgressPreference = 'Continue'            # Subsequent calls do display UI.
    # get links from actual meetings and attach $baseurl
    $page.links.href |?{($_ -like "*$Jaar*") -and ($_.Split("/").GetUpperBound(0) -gt  4) } |%{-join("$baseurl",$_)} |%{
      $url2 = $_
      $ProgressPreference = 'SilentlyContinue'    # Subsequent calls do not display UI.
        $page2 = Invoke-Webrequest $url2
      $ProgressPreference = 'Continue'            # Subsequent calls do display UI.
      $page2.links.href |?{($_ -like "*.pdf*") -or ($_ -like "*.doc*") -or ($_ -like "*.txt*") -or ($_ -like "*.xls*") -or ($_ -like "*.ppt*") -or ($_ -like "*.rtf*")} |%{-join("$baseurl",$_)} | Out-File -FilePath $output -Append
    }
	}
}
